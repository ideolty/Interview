经典问题：两个线程，一个输出字母，一个输出数字，交替输出1A2B3C……26Z

```java
    private static char[] nums = "123456789".toCharArray();
    private static char[] chars = "abcdefghi".toCharArray();
    private static Thread t2;
    private static Thread t1;

		// 使用三种常用的方法来实现
    public static void main(String[] args) {
        parkAndUnpark();
        waitAndNotify();
        lockAndCondition();
    }
```



**park/unpark**

```java
	 /**
		* unpark与park谁先谁后无所谓
		*/
		private static void parkAndUnpark(){
        t1 = new Thread(() -> {
            for (char num : nums) {
                System.out.println(num);
                LockSupport.unpark(t2);
                LockSupport.park();
            }
        });

        t2 = new Thread(() -> {
            for (char num : chars) {
                LockSupport.park();
                System.out.println(num);
                LockSupport.unpark(t1);
            }
        });

        t1.start();
        t2.start();
    }
```



**Sychornized + wait + notify**

```java
   private static void waitAndNotify(){
        Object object = new Object();
        CountDownLatch countDownLatch = new CountDownLatch(1);
        new Thread(() -> {
            synchronized (object){
                try {
                    for (char num : nums) {
                        System.out.println(num);

                        if (countDownLatch.getCount() != 0){
                            countDownLatch.countDown();
                        }

                        object.notify();
                        object.wait();
                    }
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }

                object.notify();
            }

        }).start();

        new Thread(() -> {
            try {
                countDownLatch.await();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }

            synchronized (object){
                try {
                    for (char num : chars) {
                        System.out.println(num);
                        object.notify();
                        object.wait();
                    }
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        }).start();
    }
```



**ReentrantLock + Condition**

```java
    private static void lockAndCondition(){
        Lock lock = new ReentrantLock();
        Condition condition1 = lock.newCondition();
        Condition condition2 = lock.newCondition();
        CountDownLatch countDownLatch = new CountDownLatch(1);

        t1 = new Thread(() -> {
            try {
                lock.lock();
                for (char num : nums) {
                    System.out.println(num);
                    if (countDownLatch.getCount() != 0){
                        countDownLatch.countDown();
                    }

                    try {
                        condition2.signal();
                        condition1.await();
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }

                condition2.signal();
            } catch (Exception e) {
                e.printStackTrace();
            }finally {
                lock.unlock();
            }
        });

        t2 = new Thread(() -> {
            try {
                try {
                    countDownLatch.await();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }

                lock.lock();
                for (char num : chars) {
                    try {
                        System.out.println(num);
                        condition1.signal();
                        condition2.await();
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
                condition1.signal();
            } catch (Exception e) {
                e.printStackTrace();
            } finally {
                lock.unlock();
            }
        });
        t2.start();
        t1.start();
    }
```
